# PSEUDOCODE for the challenge
------------------------------

FUNCTION create_challenge_csv(out_path="cybercon_challenge.csv", n_rows=1000, noise_std=1.0, seed=0):
    SET random seed to seed
    INITIALIZE n_features = length_of_secret_message  # do NOT include the message text

    # create vector v of feature means (ASCII codes of secret message)
    v = array_of_length(n_features)  # each element = ordinal value of each secret character

    # create matrix X of random noise
    X = random_normal_matrix(rows=n_rows, cols=n_features, mean=0.0, std=noise_std)

    # pick a random row to embed the exact (noisy) secret vector
    insert_idx = random_integer_in_range(0, n_rows-1)
    X[insert_idx] = v + small_random_noise_vector(std=0.05, length=n_features)

    # create two nearby rows that contain scaled versions of v (with small noise)
    X[(insert_idx + 1) % n_rows] = v * 0.6 + small_random_noise_vector(std=0.1, length=n_features)
    X[(insert_idx + 2) % n_rows] = v * 0.4 + small_random_noise_vector(std=0.1, length=n_features)

    # randomly permute the rows so embedded rows are shuffled among the dataset
    order = random_permutation_of(0..n_rows-1) using seed+1
    X = X[order]

    # label columns f0, f1, ..., f(n_features-1)
    columns = ["f0", "f1", ..., "f{n_features-1}"]

    # write to CSV without index
    SAVE X as CSV at out_path with header=columns, index=False

    PRINT message: file written and parameters used

END FUNCTION

# When executed as a script:
CALL create_challenge_csv(out_path='data.csv', n_rows=1000, noise_std=1.0, seed=0)
